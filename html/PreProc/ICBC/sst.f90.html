20:       program sst
21: 
22:       use mod_dynparam
23:       use mod_sst_1deg
24:       use mod_sst_eh5om
25:       use mod_sst_ersst
26:       use mod_sst_fvgcm
27: 
28:       implicit none
29:       integer :: ierr
30:       character(256) :: namelistfile , prgname
31: !
32: !     Read input global namelist
33: !
34:       call getarg(0, prgname)
35:       call getarg(1, namelistfile)
36:       call initparam(namelistfile, ierr)
37:       if ( ierr/=0 ) then
38:         write ( 6, * ) 'Parameter initialization not completed'
39:         write ( 6, * ) 'Usage : '
40:         write ( 6, * ) '          ', trim(prgname), ' regcm.in'
41:         write ( 6, * ) ' '
42:         write ( 6, * ) 'Check argument and namelist syntax'
43:         stop
44:       end if
45: 
46:       if ( ssttyp=='GISST' .or. ssttyp=='OISST' .or.                    &
47:        &   ssttyp=='OI_NC' .or. ssttyp=='OI2ST' .or.                    &
48:        &   ssttyp=='OI_WK' .or. ssttyp=='OI2WK' ) then
49:         call sst_1deg
50:       else if ( ssttyp=='EH5RF' .or. ssttyp=='EH5A2' .or.               &
51:        &        ssttyp=='EH5B1' .or. ssttyp=='EHA1B' ) then
52:         call sst_eh5om
53:       else if ( ssttyp=='ERSST' .or. ssttyp=='ERSKT' ) then
54:         call sst_ersst
55:       else if ( ssttyp=='FV_RF' .or. ssttyp=='FV_A2' .or.               &
56:        &        ssttyp=='FV_B2' ) then
57:         call sst_fvgcm
58:       else
59:         print *, 'Unknown SSTTYP ', ssttyp , ' specified in ',          &
60:               & trim(namelistfile)
61:         stop
62:       end if
63: 
64:       print *, 'Successfully generated SST'
65: 
66:       end program sst
