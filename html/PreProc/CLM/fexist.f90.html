20:       subroutine fexist(filnam)
21:       implicit none
22: !
23: ! Dummy arguments
24: !
25:       character(256) :: filnam
26:       intent (inout) filnam
27: !
28: ! Local variables
29: !
30:       logical :: there
31:       character(1) :: yesno
32: 
33:  100  continue
34:       inquire (file=filnam,exist=there)
35:       if ( there ) then
36:  150    continue
37:         print * , ' '
38:         print * , ' '
39:         print * , '**************************************************'
40:         print * , 'FILE ALREADY EXISTS:  ' , trim(filnam)
41:         print * , 'Do you want to overwrite the existing file? [y/n/q]'
42:         read (*,*) yesno
43:         if ( yesno=='y' ) then
44:           return
45:         else if ( yesno=='n' ) then
46:           print * , 'ENTER NEW FILE NAME'
47:           read (*,'(a)') filnam
48:           print * , 'USING ', trim(filnam)
49:           goto 100
50:         else if ( yesno=='q' ) then
51:           stop 999
52:         else
53:           go to 150
54:         end if
55:       end if
56:  
57:       end subroutine fexist
