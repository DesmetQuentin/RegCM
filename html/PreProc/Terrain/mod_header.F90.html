24:       subroutine header(myid)
25:       implicit none 
26: !
27: ! Dummy arguments
28: !
29:       integer , intent(in) :: myid
30: !
31: ! Local variables:
32: !
33:       integer , parameter :: nrite = 6
34:       integer :: hostnm
35:       integer :: ihost, idir
36:       integer :: getcwd
37:       character (len=32) :: cdata='?'
38:       character (len=32) :: hostname='?' 
39:       character (len=32) :: user='?' 
40:       character (len=128) :: directory='?'
41: !
42:       if (myid.eq.1)  then 
43:         write (nrite,"(/,2x,'This is RegCM version 4 ')")
44:         write (nrite,100)  SVN_REV, __DATE__ , __TIME__   
45: 100     format(2x,' SVN Revision: ',a,' compiled at: data : ',          &
46:            &   a,'  time: ',a,/)
47: 
48: #ifdef IBM
49:         hostname='ibm platform '
50:         user= 'Unknown'
51:         call fdate_(cdata)
52: #else
53:         Ihost = hostnm(hostname)
54:         call getlog(user)
55:         call fdate(cdata)
56: #endif 
57: 
58:         Idir=getcwd(directory)
59: 
60:         write(nrite,*) ": this run start at  : ",trim(cdata)
61:         write(nrite,*) ": it is submitted by : ",trim(user)
62:         write(nrite,*) ": it is running on   : ",trim(hostname)
63:         write(nrite,*) ": in directory       : ",trim(directory)
64:         write(nrite,*) "                     " 
65:       end if 
66:       end subroutine header
67: 
68:       end module mod_header
