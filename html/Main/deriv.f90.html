20:       subroutine deriv
21: 
22: !:::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
23: !
24: !     derivatives of energy fluxes with respect to leaf temperature for
25: !     newton-raphson calculation of leaf temperature.
26: !     input: rs,ra,cdrd,rppq,efe.    output: qsatld,dcd.
27: !
28: !     approximate by derivatives of cdr and ef.  many weaker
29: !     dependences on leaf temperature are omitted, as convergence
30: !     rate is not affected.
31: !
32:       use mod_dynparam
33:       use mod_bats , only : wtg0 , ldoc1d , sigf , tlef1d , df , cdr ,  &
34:                     & ts1d , wtga , qsatl , wta0 , wtgaq , tg1d , cn1
35:       use mod_constants , only : tzero
36:       use mod_ictp01
37:       implicit none
38: !
39: ! Local variables
40: !
41:       real(8) :: dne , hfl , xkb
42:       integer :: n , i
43: !
44:       do i = 1 , iym1
45:         do n = 1 , nnsg
46:           if ( ldoc1d(n,i).gt.0.5 ) then
47:             if ( sigf(n,i).gt.0.001 ) then
48:               dne = 1./(tlef1d(n,i)-b(n,i))
49:               qsatld(n,i) = qsatl(n,i)*a(n,i)*(tzero-b(n,i))*dne**2
50:               xkb = cdrd(n,i)/cdr(n,i)
51:               hfl = df(n,i)*(wtga(n,i)*tlef1d(n,i)-wtg0(n,i)*tg1d(n,i)  &
52:                   & -wta0(n,i)*ts1d(n,i))
53:               dcd(n,i) = cn1(n,i)*rppq(n,i)*wtgaq(n,i)*qsatld(n,i)      &
54:                         & + (1.-wtgaq(n,i))*efe(n,i)                    &
55:                         & *xkb + (1.-wtga(n,i))*hfl*xkb
56:               dcd(n,i) = dmax1(dcd(n,i),0.D0)
57:               dcd(n,i) = dmin1(dcd(n,i),500.D0)
58:             end if
59:           end if
60:         end do
61:       end do
62:  
63:       end subroutine deriv
