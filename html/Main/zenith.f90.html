20:       subroutine zenith(dec,alat,fjd,coszrs,frac,imax)
21: !
22: !:::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
23: !
24: ! ** used by radiation package in ccm to coszrs  *****not yet used
25: !                 here but could be called from albedo
26: !
27: !     this routine computes the zenith angle at each of 'imax'
28: !     equally-spaced longitudes at latitude 'alat', for a diurnally
29: !     varying sun.
30: !
31: !     input:  dec    - declination of sun in radians. computed in
32: !                      'solar'.
33: !             alat   - latitude in radians.
34: !             fjd    - fractional julian date in days.  the code conven-
35: !                      tion (see 'compjd') is that fjd=0 at noon  at
36: !                      greenwich meridian (1200 gmt), so the hour angle
37: !                      at longitude 'lon' is ha=lon+twopi*fjd.
38: !             imax   - number of equally spaced longitude values.
39: !                      it is assumed that i=1 and i=imax+1 are both
40: !                      at the greenwich meridian (lon=0).
41: !     output: coszrs - cos(z)/ where 'z' is the zenith angle at
42: !                      longitude 'i', latitude 'alat', and time 'fjd'.
43: !                           cos(z)=sin(alat)*sin(dec)+
44: !                                  cos(alat)*cos(dec)*cos(ha)
45: !                      it is assumed that the
46: !                      annual mean solar constant is used elsewhere
47: !                      in determining the solar flux.
48: !                      the 1/r**2 decrease of the solar flux appears
49: !                      in subroutine radiatn as eccf
50: !             frac   - not used in diurnal mode: set to 1.  in the
51: !                      average insolation mode, 'frac' is the daylight
52: !                      fraction at the point (see 'zenith'); to lowest
53: !                      order it should be independent of longitude.
54: !
55: !:::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
56: !
57:       use mod_constants , only : twopi
58: 
59:       implicit none
60: !
61: ! Dummy arguments
62: !
63:       integer , intent (in) :: imax
64:       real(kind=8) , intent (in) :: alat , dec , fjd
65:       real(kind=8) , intent (out) , dimension(imax) :: coszrs , frac
66: !
67: ! Local variables
68: !
69:       real(8) :: cc , cosz , dlon , ha , ss , tpifjd
70:       integer :: i
71: !
72: !***********************************************************************
73: !
74:       ss = dsin(alat)*dsin(dec)
75:       cc = dcos(alat)*dcos(dec)
76:       dlon = twopi/imax
77:       tpifjd = twopi*fjd
78:       do i = 1 , imax
79:         frac(i) = 1.0
80:         ha = (i-1)*dlon + tpifjd
81: !       if cosz is negative, the sun is below the horizon.
82:         cosz = dmax1(0.D0,ss+cc*dcos(ha))
83:         coszrs(i) = cosz
84:       end do
85: !
86:       end subroutine zenith
