20:       subroutine ljchrn(chrv,nlshf)
21: !
22: !****     routine left justifies a character variable cnull filling on
23: !****     the right.  chrv is the character variable.  nlshf tells
24: !****     how many characters the variable was shifted left
25: !
26:       implicit none
27: !
28: ! Dummy arguments
29: !
30:       character(*) :: chrv
31:       integer :: nlshf
32:       intent (inout) nlshf
33: !
34: ! Local variables
35: !
36:       character(1) :: blank , cnull
37:       integer :: lnchrv
38: !
39:       data blank/' '/
40: !
41:       cnull = char(0)
42:       lnchrv = len(chrv)
43:       nlshf = 0
44:       do while ( chrv(1:1).eq.blank .or. chrv(1:1).eq.cnull )
45:         call lshfch(chrv,lnchrv,1,1)
46:         nlshf = nlshf + 1
47:         if ( nlshf.eq.lnchrv ) exit
48:       end do
49:       end subroutine ljchrn
