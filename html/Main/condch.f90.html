20:       subroutine condch
21: 
22: !:::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
23: !
24: !     dimensional and non-dimensional sensible heat conductances
25: !               for canopy and soil flux calculations
26: !
27:       use mod_dynparam
28:       use mod_bats , only : vegt , vspda , cdr , lveg , sigf , uaf ,    &
29:                     & cf , wta , wtlh , wtg , wtshi , wtl0 , wtg0 ,     &
30:                     & wtgl , wta0 , wtga , ldoc1d , sqrtdi
31:       use mod_constants , only : csoilc
32:       implicit none
33: !
34: ! Local variables
35: !
36:       integer :: n , i
37: !
38: !     csoilc = constant drag coefficient for soil under canopy;
39: !     prescribed in subroutine bconst.
40: !
41: !     symbols used for weights are:   wt : weight
42: !     a : air
43: !     l : leaf
44: !     i : inverse
45: !     s : sum
46: !     h : sensible heat
47: !     q : water vapor
48: !     0 : normalized (sums to one)
49: !     g : ground
50: !
51:       do i = 2 , iym1
52:         do n = 1 , nnsg
53:           if ( ldoc1d(n,i).gt.0.5 ) then
54:             if ( sigf(n,i).gt.0.001 ) then
55:               uaf(n,i) = vspda(n,i)*dsqrt(cdr(n,i))
56:               cf(n,i) = 0.01*sqrtdi(lveg(n,i))/dsqrt(uaf(n,i))
57:               wta(n,i) = sigf(n,i)*cdr(n,i)*vspda(n,i)
58:               wtlh(n,i) = cf(n,i)*uaf(n,i)*vegt(n,i)
59:               wtg(n,i) = csoilc*uaf(n,i)*sigf(n,i)
60:               wtshi(n,i) = 1./(wta(n,i)+wtlh(n,i)+wtg(n,i))
61:               wtl0(n,i) = wtlh(n,i)*wtshi(n,i)
62:               wtg0(n,i) = wtg(n,i)*wtshi(n,i)
63:               wtgl(n,i) = wtl0(n,i) + wtg0(n,i)
64:               wta0(n,i) = 1. - wtgl(n,i)
65:               wtga(n,i) = wta0(n,i) + wtg0(n,i)
66:             end if
67:           end if
68:         end do
69:       end do
70: !
71:       end subroutine condch
