20:       subroutine frawat
21: 
22: !:::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
23: !
24: !  **** determines fraction of foliage covered by water fwet, and
25: !  **** the fraction of foliage that is dry transpiring leaf fdry.
26: !  note: their defns differ - fwet is the fraction of all veg surfaces
27: !  which are wet because mod_stems can evaporate, fdry is the fraction
28: !  of lai which is dry because only leaves can transpire
29: !
30: !  ldew1d(i) is in kg/m**2/s
31: !  fwet   = ratio of dew to max value to 2/3 power
32: !           ( 2/3 power comes from deardorff (1978) )
33: !              ** keep fwet le 1.0 **
34: !  dewmxi = inverse of max allowed dew depth on leaf in mm
35: !
36:       use mod_dynparam
37:       use mod_bats , only : xlai , ldoc1d , sigf , ldew1d , fwet ,      &
38:                    & xlsai , fdry , vegt
39:       use mod_constants , only : dewmxi
40:       implicit none
41: !
42: ! Local variables
43: !
44:       integer :: n , i
45: !
46:       do i = 2 , iym1
47:         do n = 1 , nnsg
48:           if ( ldoc1d(n,i).gt.0.5 ) then
49:             if ( sigf(n,i).gt.0.001 ) then
50:               fwet(n,i) = 0.
51:               if ( ldew1d(n,i).gt.0. ) then
52:                 fwet(n,i) = ((dewmxi/vegt(n,i))*ldew1d(n,i))**(2.0/3.0)
53:                 fwet(n,i) = dmin1(fwet(n,i),1.D0)
54:               end if
55:               fdry(n,i) = (1.-fwet(n,i))*xlai(n,i)/xlsai(n,i)
56:             end if
57:           end if
58:         end do
59:       end do
60:  
61:       end subroutine frawat
