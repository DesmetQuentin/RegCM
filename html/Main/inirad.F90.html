20:       subroutine inirad
21:  
22:       use mod_dynparam
23:       use mod_o3blk , only : o3data
24:       use mod_rad , only : o3prof , heatrt
25:       use mod_date , only : jyear , jyear0 , ktau
26:       implicit none
27: !
28: ! Local variables
29: !
30:       integer :: i , j , k
31: !
32: !     compute ozone mixing ratio distribution
33: !
34:       if ( jyear.eq.jyear0 .and. ktau.eq.0 ) then
35:         do k = 1 , kz
36: #ifdef MPP1
37:           do j = 1 , jendl
38:             do i = 1 , iym1
39:               heatrt(i,k,j) = 0.
40:               o3prof(i,k,j) = 0.
41:             end do
42:           end do
43: #else
44:           do j = 1 , jxm1
45:             do i = 1 , iym1
46:               heatrt(i,k,j) = 0.
47:               o3prof(i,k,j) = 0.
48:             end do
49:           end do
50: #endif
51:         end do
52: #ifdef MPP1
53:         do j = 1 , jendl
54:           do i = 1 , iym1
55:             o3prof(i,kzp1,j) = 0.
56:           end do
57:         end do
58: #else
59:         do j = 1 , jxm1
60:           do i = 1 , iym1
61:             o3prof(i,kzp1,j) = 0.
62:           end do
63:         end do
64: #endif
65:         call o3data
66: #ifdef MPP1
67:         if ( myid.eq.0 ) then
68:           print * , 'ozone profiles'
69:           do k = 1 , kzp1
70:             write (6,99001) o3prof(3,k,jxp)
71:           end do
72:         end if
73: #else
74:         print * , 'ozone profiles'
75:         do k = 1 , kzp1
76:           write (6,99001) o3prof(3,k,3)
77:         end do
78: #endif
79:       end if
80: 99001 format (1x,7E12.4)
81:  
82:       end subroutine inirad
