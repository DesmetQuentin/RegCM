20:       subroutine drip
21: !
22: !  excess leaf water is put into rain or snow;
23: !  leaf water is reset to its maximum value.
24: !
25:       use mod_dynparam
26:       use mod_bats , only : etrrun , ldew1d , vegt , sdrop , xrun ,     &
27:                     & ldoc1d , sigf , tm
28:       use mod_constants , only : dewmx , tzero
29:       implicit none
30: !
31: ! Local variables
32: !
33:       integer :: n , i
34: !
35:       do i = 2 , iym1
36:         do n = 1 , nnsg
37:           if ( ldoc1d(n,i).gt.0.5 ) then
38:             if ( sigf(n,i).gt.0.001 ) then
39: !             ***********         xrun = leaf drip ; sdrop = snow drop
40: !             off foliage
41:               etrrun(n,i) = 0.
42:               xrun(n,i) = ldew1d(n,i) - dewmx*vegt(n,i)
43:               sdrop(n,i) = 0.
44:  
45: !             ***********         test on maximum value of dew
46:               if ( xrun(n,i).gt.0. ) then
47:                 etrrun(n,i) = xrun(n,i)
48:                 ldew1d(n,i) = dewmx*vegt(n,i)
49:               end if
50:  
51: !             ***********         below freezing excess leaf water
52: !             falls as snow
53:               if ( (xrun(n,i).gt.0.) .and. (tm(n,i).lt.tzero) ) then
54:                 etrrun(n,i) = 0.
55:                 sdrop(n,i) = xrun(n,i)
56:               end if
57:             end if
58:           end if
59:         end do
60:       end do
61:  
62:       end subroutine drip
