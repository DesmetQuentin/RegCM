20:       subroutine dragdn
21: !
22: !     returns neutral drag coefficient for grid square
23: !
24: !     zlnd = soil roughness length
25: !     zoce = ocean roughness length
26: !     zsno = snow roughness length
27: !     vonkar = von karman constant
28: !
29: !     frav = fraction of grid point covered by vegetation
30: !     fras = fraction of grid point covered by snow
31: !     frab = fraction of grid point covered by bare soil
32: !     cdb = neutral drag coeff over bare soil, ocean, sea ice
33: !     cds = neutral drag coeff over snow
34: !     cdv = neutral drag coeff over vegetation
35: !     cdrn = neutral drag coeff for momentum avgd over grid point
36: !
37:       use mod_dynparam
38:       use mod_bats , only : displa , lveg , z1d , z1 , z1log , cdrn ,   &
39:                    & sigf , sice1d , ldoc1d , veg1d , scvk , rough , wt
40:       use mod_constants , only : zlnd , zoce , zsno , vonkar
41:       implicit none
42: !
43: ! Local variables
44: !
45:       real(8) :: asigf , cdb , cds , cdv , frab , fras , frav
46:       integer :: n , i
47: !
48: !     ******           sea ice classified same as desert
49:       do i = 2 , iym1
50:         do n = 1 , nnsg
51:           if ( lveg(n,i).le.0 .and. sice1d(n,i).gt.0. ) lveg(n,i) = 8
52:         end do
53:       end do
54:  
55:       call depth
56:  
57:       do i = 2 , iym1
58:         do n = 1 , nnsg
59:  
60:           z1(n,i) = z1d(n,i)
61:           z1log(n,i) = dlog(z1(n,i))
62:  
63: #ifdef SEAICE
64:            if ( ldoc1d(n,i).gt.1.5 ) then
65:              sigf(n,i) = 0.0
66:              cdrn(n,i) = ( vonkar / dlog( z1(n,i)/zlnd ) )**2
67:            else if ( ldoc1d(n,i).gt.0.5 ) then
68: #else
69:            if ( ldoc1d(n,i).gt.0.5 ) then
70: #endif
71:  
72: !           ******           drag coeff over land
73:             frav = sigf(n,i)
74:             asigf = veg1d(n,i)
75:             fras = asigf*wt(n,i) + (1.-asigf)*scvk(n,i)
76:             frab = (1.-asigf)*(1.-scvk(n,i))
77:             cdb = (vonkar/dlog(z1(n,i)/zlnd))**2
78:             cds = (vonkar/dlog(z1(n,i)/zsno))**2
79:             cdv = (vonkar/dlog((z1(n,i)-displa(lveg(n,i)))/             &
80:                 & rough(lveg(n,i))))**2
81:             cdrn(n,i) = frav*cdv + frab*cdb + fras*cds
82:  
83:           else
84: !           ******           drag coeff over ocean
85:             sigf(n,i) = 0.0
86:             cdrn(n,i) = (vonkar/dlog(z1(n,i)/zoce))**2
87:           end if
88:         end do
89:       end do
90:  
91:       end subroutine dragdn
