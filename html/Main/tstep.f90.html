20:       subroutine tstep(extime,dtinc,deltmx)
21: 
22:       use mod_dynparam
23:       use mod_param1 , only : dt , dt2 , dtmin
24:       use mod_date , only : jyear , jyear0 , ktau
25:       implicit none
26: 
27: !cccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccc
28: !                                                                     c
29: !                                                                     c
30: !     this subroutine makes a refined start to the model, i.e. the    c
31: !     model is started with a series of small time steps.             c
32: !                                                                     c
33: !                                                                     c
34: !cccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccc
35: !
36: ! Dummy arguments
37: !
38:       real(8) :: deltmx , dtinc , extime
39:       intent (in) extime
40:       intent (out) dtinc
41:       intent (inout) deltmx
42: !
43: ! Local variables
44: !
45:       real(8) :: deltmn , tscale
46:       integer :: idtmax
47: !
48: !---------------------------------------------------------------------
49: !
50:       deltmn = 0.0
51:       tscale = 0.0
52:       if ( jyear.eq.jyear0 .and. ktau.eq.0 ) then
53:         idtmax = 1
54:         tscale = 5.*dt
55:         deltmn = 0.1*dt
56:         deltmx = dt
57:       end if
58:       if ( idtmax.eq.1 ) then
59:         dt = deltmx*(1.-dexp(-extime/tscale)) + deltmn
60:         dtinc = dt
61:         dtmin = dt/60.
62:         idtmax = 2
63:       end if
64:       dt2 = 2.*dt
65:       if ( jyear.ne.jyear0 .or. ktau.ne.0 ) dt = dt2
66: !
67:       end subroutine tstep
