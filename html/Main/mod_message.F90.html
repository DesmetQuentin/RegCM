63:       subroutine fatal(filename,line,str)
64: #ifdef MPP1
65:       use mpi
66: #endif
67:       implicit none
68: !
69: ! Dummy arguments
70: !
71:       character(*) :: filename , str
72:       integer :: line
73:       intent (in) filename , line , str
74: !
75: ! Local variables
76: !
77:       character(8) :: cline
78: #ifdef MPP1
79:       integer :: ierr
80: #endif
81: !
82:       write (cline,'(i6)') line
83:       write (aline,*) '-------------- FATAL CALLED ---------------'
84:       call say
85:       if ( line.gt.0 ) then
86:         write (aline,*) 'Fatal in file: '//filename//' at line: '//     &
87:                       & trim(cline)
88:         call say
89:       end if
90:       write (aline,*) str
91:       call say
92:       write (aline,*) '-------------------------------------------'
93:       call say
94: #ifdef MPP1
95:       call mpi_abort(mpi_comm_world,1,ierr)
96: #else
97:       call abort
98: #endif
99:       end subroutine fatal
