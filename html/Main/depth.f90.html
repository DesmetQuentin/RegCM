20:       subroutine depth
21: !
22: !          wt = fraction of vegetation covered by snow
23: !        sigf = fraction of veg. cover, excluding snow-covered veg.
24: !        scvk = fraction of soil covered by snow
25: !
26: !     scrat = snow depth (m) =  .001 snow depth (mm) / snow density
27: !     rhosw = ratio of snow density to density of h2o
28: !
29: !     height of vegetation assumed to be 10 x vegetation roughness ht
30: !     densi is defined the same here as in subr. albedo
31: !
32: !     wt now scaled so that runs betw. 0 & 1 and wt=0.5 when depth
33: !     of snow equals height of vegetation
34: !
35:  
36:       use mod_dynparam
37:       use mod_bats , only : lveg , rhosw , densi , scrat , wt , rough , &
38:                     & veg1d , sigf , scvk , ldoc1d , sag1d , scv1d
39:       implicit none
40: !
41: ! Local variables
42: !
43:       real(8) :: age
44:       integer :: n , i
45: ! 
46:       do i = 2 , iym1
47:         do n = 1 , nnsg
48:           if ( ldoc1d(n,i).gt.0.5 ) then
49:             age = (1.-1./(1.+sag1d(n,i)))
50:             rhosw(n,i) = .10*(1.+3.*age)
51:             densi(n,i) = .01/(1.+3.*age)
52:             scrat(n,i) = scv1d(n,i)*densi(n,i)
53:             wt(n,i) = 1.0
54:             if ( lveg(n,i).gt.0 ) then
55:               wt(n,i) = 0.1*scrat(n,i)/rough(lveg(n,i))
56:               wt(n,i) = wt(n,i)/(1.+wt(n,i))
57:             end if
58:             sigf(n,i) = (1.-wt(n,i))*veg1d(n,i)
59:             scvk(n,i) = scrat(n,i)/(0.1+scrat(n,i))
60:           end if
61:         end do
62:       end do
63:  
64:       end subroutine depth
