20:       subroutine trcems(k,co2t,pnm,ucfc11,ucfc12,un2o0,un2o1,bn2o0,     &
21:                       & bn2o1,uch4,bch4,uco211,uco212,uco213,uco221,    &
22:                       & uco222,uco223,uptype,w,s2c,up2,emplnk,th2o,tco2,&
23:                       & to3,emstrc)
24: !
25:       use mod_dynparam
26:       use mod_constants , only : sslp
27:       implicit none
28: !
29: !----------------------------------------------------------------------
30: !  Calculate emissivity for CH4, N2O, CFC11 and CFC12 bands.
31: !            Coded by J.T. Kiehl November 21, 1994
32: !-----------------------------------------------------------------------
33: !
34: !------------------------------Arguments--------------------------------
35: !
36: ! co2t   - pressure weighted temperature
37: ! pnm    - interface pressure
38: ! ucfc11 - CFC11 path length
39: ! ucfc12 - CFC12 path length
40: ! un2o0  - N2O path length
41: ! un2o1  - N2O path length (hot band)
42: ! uch4   - CH4 path length
43: ! uco211 - CO2 9.4 micron band path length
44: ! uco212 - CO2 9.4 micron band path length
45: ! uco213 - CO2 9.4 micron band path length
46: ! uco221 - CO2 10.4 micron band path length
47: ! uco222 - CO2 10.4 micron band path length
48: ! uco223 - CO2 10.4 micron band path length
49: ! uptype - continuum path length
50: ! bn2o0  - pressure factor for n2o
51: ! bn2o1  - pressure factor for n2o
52: ! bch4   - pressure factor for ch4
53: ! emplnk - emissivity Planck factor
54: ! th2o   - water vapor overlap factor
55: ! tco2   - co2 overlap factor
56: ! to3    - o3 overlap factor
57: ! s2c    - h2o continuum path length
58: ! w      - h2o path length
59: ! up2    - pressure squared h2o path length
60: ! k      - level index
61: !
62: !------------------------------Output Arguments-------------------------
63: !
64: ! emstrc - total trace gas emissivity
65: !
66: !-----------------------------------------------------------------------
67: !
68: ! Dummy arguments
69: !
70:       integer :: k
71:       real(8) , dimension(iym1,kzp1) :: bch4 , bn2o0 , bn2o1 , co2t , &
72:            & emstrc , pnm , s2c , ucfc11 , ucfc12 , uch4 , uco211 ,     &
73:            & uco212 , uco213 , uco221 , uco222 , uco223 , un2o0 ,       &
74:            & un2o1 , uptype , w
75:       real(8) , dimension(14,iym1) :: emplnk
76:       real(8) , dimension(iym1) :: tco2 , th2o , to3 , up2
77:       intent (in) bch4 , bn2o0 , bn2o1 , co2t , emplnk , k , pnm , s2c ,&
78:                 & tco2 , th2o , to3 , ucfc11 , ucfc12 , uch4 , uco211 , &
79:                 & uco212 , uco213 , uco221 , uco222 , uco223 , un2o0 ,  &
80:                 & un2o1 , up2 , uptype , w
81:       intent (out) emstrc
82: !
83: ! Local variables
84: !
85: ! sqti   - square root of mean temp
86: ! ecfc1  - emissivity of cfc11 798 cm-1 band
87: ! ecfc2  -     "      "    "   846 cm-1 band
88: ! ecfc3  -     "      "    "   933 cm-1 band
89: ! ecfc4  -     "      "    "   1085 cm-1 band
90: ! ecfc5  -     "      "  cfc12 889 cm-1 band
91: ! ecfc6  -     "      "    "   923 cm-1 band
92: ! ecfc7  -     "      "    "   1102 cm-1 band
93: ! ecfc8  -     "      "    "   1161 cm-1 band
94: ! u01    - n2o path length
95: ! u11    - n2o path length
96: ! beta01 - n2o pressure factor
97: ! beta11 - n2o pressure factor
98: ! en2o1  - emissivity of the 1285 cm-1 N2O band
99: ! u02    - n2o path length
100: ! u12    - n2o path length
101: ! beta02 - n2o pressure factor
102: ! en2o2  - emissivity of the 589 cm-1 N2O band
103: ! u03    - n2o path length
104: ! beta03 - n2o pressure factor
105: ! en2o3  - emissivity of the 1168 cm-1 N2O band
106: ! betac  - ch4 pressure factor
107: ! ech4   - emissivity of 1306 cm-1 CH4 band
108: ! betac1 - co2 pressure factor
109: ! betac2 - co2 pressure factor
110: ! eco21  - emissivity of 1064 cm-1 CO2 band
111: ! eco22  - emissivity of 961 cm-1 CO2 band
112: ! tt     - temp. factor for h2o overlap factor
113: ! psi1   - narrow band h2o temp. factor
114: ! phi1   -            "
115: ! p1     - h2o line overlap factor
116: ! w1     -          "
117: ! tw     - h2o transmission overlap
118: ! g1     - h2o overlap factor
119: ! g2     -          "
120: ! g3     -          "
121: ! g4     -          "
122: ! ab     -          "
123: ! bb     -          "
124: ! abp    -          "
125: ! bbp    -          "
126: ! tcfc3  - transmission for cfc11 band
127: ! tcfc4  -         "
128: ! tcfc6  - transmission for cfc12 band
129: ! tcfc7  -          "
130: ! tcfc8  -          "
131: ! tlw    - h2o overlap factor
132: ! tch4   - ch4 overlap factor
133: !
134:       real(8) , dimension(6) :: ab , abp , bb , bbp , g1 , g2 , g3 , g4
135:       real(8) :: b , beta01 , beta02 , beta03 , beta11 , betac ,        &
136:                & betac1 , betac2 , ecfc1 , ecfc2 , ecfc3 , ecfc4 ,      &
137:                & ecfc5 , ecfc6 , ecfc7 , ecfc8 , ech4 , eco21 , eco22 , &
138:                & en2o1 , en2o2 , en2o3 , p1 , phi1 , psi1 , tcfc3 ,     &
139:                & tcfc4 , tcfc6 , tcfc7 , tcfc8 , tch4 , tlw , u , u01 , &
140:                & u02 , u03 , u11 , u12 , w1
141:       real(8) :: func
142:       integer :: i , l
143:       real(8) , dimension(iym1) :: sqti , tt
144:       real(8) , dimension(iym1,6) :: tw
145: !
146:       data g1/0.0468556D0 , 0.0397454D0 , 0.0407664D0 , 0.0304380D0 ,   &
147:          & 0.0540398D0 , 0.0321962D0/
148:       data g2/14.4832D0 , 4.30242D0 , 5.23523D0 , 3.25342D0 ,           &
149:          & 0.698935D0 , 16.5599D0/
150:       data g3/26.1898D0 , 18.4476D0 , 15.3633D0 , 12.1927D0 ,           &
151:          & 9.14992D0 , 8.07092D0/
152:       data g4/0.0261782D0 , 0.0369516D0 , 0.0307266D0 , 0.0243854D0 ,   &
153:          & 0.0182932D0 , 0.0161418D0/
154:       data ab/3.0857D-2 , 2.3524D-2 , 1.7310D-2 , 2.6661D-2 ,           &
155:          & 2.8074D-2 , 2.2915D-2/
156:       data bb/ - 1.3512D-4 , -6.8320D-5 , -3.2609D-5 , -1.0228D-5 ,     &
157:          & -9.5743D-5 , -1.0304D-4/
158:       data abp/2.9129D-2 , 2.4101D-2 , 1.9821D-2 , 2.6904D-2 ,          &
159:          & 2.9458D-2 , 1.9892D-2/
160:       data bbp/ - 1.3139D-4 , -5.5688D-5 , -4.6380D-5 , -8.0362D-5 ,    &
161:          & -1.0115D-4 , -8.8061D-5/
162:       func(u,b) = u/dsqrt(4.0+u*(1.0+1.0/b))
163: !
164:       do i = 1 , iym1
165:         sqti(i) = dsqrt(co2t(i,k))
166: !       Transmission for h2o
167:         tt(i) = dabs(co2t(i,k)-250.0)
168:       end do
169: !
170:       do l = 1 , 6
171:         do i = 1 , iym1
172:           psi1 = dexp(abp(l)*tt(i)+bbp(l)*tt(i)*tt(i))
173:           phi1 = dexp(ab(l)*tt(i)+bb(l)*tt(i)*tt(i))
174:           p1 = pnm(i,k)*(psi1/phi1)/sslp
175:           w1 = w(i,k)*phi1
176:           tw(i,l) = dexp(-g1(l)*p1*(dsqrt(1.0+g2(l)*(w1/p1))-1.0)-g3(l) &
177:                   & *s2c(i,k)-g4(l)*uptype(i,k))
178:         end do
179:       end do
180: !
181:       do i = 1 , iym1
182: !       transmission due to cfc bands
183:         tcfc3 = dexp(-175.005*ucfc11(i,k))
184:         tcfc4 = dexp(-1202.18*ucfc11(i,k))
185:         tcfc6 = dexp(-5786.73*ucfc12(i,k))
186:         tcfc7 = dexp(-2873.51*ucfc12(i,k))
187:         tcfc8 = dexp(-2085.59*ucfc12(i,k))
188: !       Emissivity for CFC11 bands
189:         ecfc1 = 50.0*(1.0-dexp(-54.09*ucfc11(i,k)))*tw(i,1)*emplnk(7,i)
190:         ecfc2 = 60.0*(1.0-dexp(-5130.03*ucfc11(i,k)))*tw(i,2)           &
191:               & *emplnk(8,i)
192:         ecfc3 = 60.0*(1.0-tcfc3)*tw(i,4)*tcfc6*emplnk(9,i)
193:         ecfc4 = 100.0*(1.0-tcfc4)*tw(i,5)*emplnk(10,i)
194: !       Emissivity for CFC12 bands
195:         ecfc5 = 45.0*(1.0-dexp(-1272.35*ucfc12(i,k)))*tw(i,3)           &
196:               & *emplnk(11,i)
197:         ecfc6 = 50.0*(1.0-tcfc6)*tw(i,4)*emplnk(12,i)
198:         ecfc7 = 80.0*(1.0-tcfc7)*tw(i,5)*tcfc4*emplnk(13,i)
199:         ecfc8 = 70.0*(1.0-tcfc8)*tw(i,6)*emplnk(14,i)
200: !       Emissivity for CH4 band 1306 cm-1
201:         tlw = dexp(-1.0*dsqrt(up2(i)))
202:         betac = bch4(i,k)/uch4(i,k)
203:         ech4 = 6.00444*sqti(i)*dlog(1.0+func(uch4(i,k),betac))          &
204:              & *tlw*emplnk(3,i)
205:         tch4 = 1.0/(1.0+0.02*func(uch4(i,k),betac))
206: !       Emissivity for N2O bands
207:         u01 = un2o0(i,k)
208:         u11 = un2o1(i,k)
209:         beta01 = bn2o0(i,k)/un2o0(i,k)
210:         beta11 = bn2o1(i,k)/un2o1(i,k)
211: !       1285 cm-1 band
212:         en2o1 = 2.35558*sqti(i)                                         &
213:               & *dlog(1.0+func(u01,beta01)+func(u11,beta11))            &
214:               & *tlw*tch4*emplnk(4,i)
215:         u02 = 0.100090*u01
216:         u12 = 0.0992746*u11
217:         beta02 = 0.964282*beta01
218: !       589 cm-1 band
219:         en2o2 = 2.65581*sqti(i)                                         &
220:               & *dlog(1.0+func(u02,beta02)+func(u12,beta02))*tco2(i)    &
221:               & *th2o(i)*emplnk(5,i)
222:         u03 = 0.0333767*u01
223:         beta03 = 0.982143*beta01
224: !       1168 cm-1 band
225:         en2o3 = 2.54034*sqti(i)*dlog(1.0+func(u03,beta03))*tw(i,6)      &
226:               & *tcfc8*emplnk(6,i)
227: !       Emissivity for 1064 cm-1 band of CO2
228:         betac1 = 2.97558*pnm(i,k)/(sslp*sqti(i))
229:         betac2 = 2.0*betac1
230:         eco21 = 3.7571*sqti(i)                                          &
231:               & *dlog(1.0+func(uco211(i,k),betac1)+func(uco212(i,k),    &
232:               & betac2)+func(uco213(i,k),betac2))*to3(i)*tw(i,5)        &
233:               & *tcfc4*tcfc7*emplnk(2,i)
234: !       Emissivity for 961 cm-1 band
235:         eco22 = 3.8443*sqti(i)                                          &
236:               & *dlog(1.0+func(uco221(i,k),betac1)+func(uco222(i,k),    &
237:               & betac1)+func(uco223(i,k),betac2))*tw(i,4)               &
238:               & *tcfc3*tcfc6*emplnk(1,i)
239: !       total trace gas emissivity
240:         emstrc(i,k) = ecfc1 + ecfc2 + ecfc3 + ecfc4 + ecfc5 + ecfc6 +   &
241:                     & ecfc7 + ecfc8 + en2o1 + en2o2 + en2o3 + ech4 +    &
242:                     & eco21 + eco22
243:       end do
244:       end subroutine trcems
