20:       subroutine slice1D(j)
21:  
22:       use mod_dynparam
23:       use mod_main
24:       use mod_pbldim
25:       use mod_bats
26:       use mod_constants , only : rgti
27:       implicit none
28: !
29: ! Dummy arguments
30: !
31:       integer :: j
32:       intent (in) j
33: !
34: ! Local variables
35: !
36:       real(8) :: amxtem , sfac
37:       integer :: n , i
38: ! 
39: !     ********* For albedov
40: !     **********            This is taken from subroutine interf so that
41: !     **********            radiation can be called in tend (not
42: !     vecbats).
43:       do i = 2 , iym1
44:         do n = 1 , nnsg
45:           ldoc1d(n,i) = ocld2d(n,i,j)
46:           sice1d(n,i) = sice2d(n,i,j)
47:           tgb1d(n,i) = tgb2d(n,i,j)
48:           ssw1d(n,i) = ssw2d(n,i,j)
49:           lveg(n,i) = nint(veg2d1(n,i,j))
50:           amxtem = dmax1(298.-tgb1d(n,i),0.D0)
51:           sfac = 1. - dmax1(0.D0,1.-0.0016*amxtem**2)
52:           if ( lveg(n,i).eq.0 ) then
53:             veg1d(n,i) = 0.
54:           else
55:             veg1d(n,i) = vegc(lveg(n,i)) - seasf(lveg(n,i))*sfac
56:           end if
57:           ts1d(n,i) = thx3d(i,kz,j)-6.5E-3*rgti*(ht1(n,i,j)-ht(i,j))
58:           scv1d(n,i) = scv2d(n,i,j)
59:           sag1d(n,i) = sag2d(n,i,j)
60:         end do
61:       end do
62:  
63:       end subroutine slice1D
