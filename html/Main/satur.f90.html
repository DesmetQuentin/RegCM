20:       subroutine satur(qsat,t,p)
21: 
22: !:::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
23: !     ****  calculates saturation vapor pressure (eg)
24: !           and qsat = saturated specific humidity (dimensionless)
25: !
26: !           uses tetens formula (1930) (ref. riegel,1974,jam,p606
27: !                                                 equation 1)
28: !:::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::::
29: !
30:       use mod_dynparam
31:       use mod_ictp01
32:       use mod_bats
33:       use mod_constants , only : tzero , c1es , c4ies , c4les , c3ies , &
34:                    &             c3les , ep2
35:       implicit none
36: !
37: ! Dummy arguments
38: !
39:       real(8) , dimension(nnsg,iym1) :: p , qsat , t
40:       intent (in) p , t
41:       intent (out) qsat
42: !
43: ! Local variables
44: !
45:       integer :: n , i
46: !
47:       do i = 2 , iym1
48:         do n = 1 , nnsg
49:           if ( t(n,i).le.tzero ) then
50:             a(n,i) = c3ies
51:             b(n,i) = c4ies
52:           else
53:             a(n,i) = c3les
54:             b(n,i) = c4les
55:           end if
56:           eg(n,i) = c1es*dexp(a(n,i)*(t(n,i)-tzero)/(t(n,i)-b(n,i)))
57:           qsat(n,i) = ep2*eg(n,i)/(p(n,i)-0.378D0*eg(n,i))
58:         end do
59:       end do
60:  
61:       end subroutine satur
