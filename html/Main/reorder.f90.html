20:       subroutine reorder(fdp,fsp,jx,iy,nz)
21:  
22:       implicit none
23: !
24: ! Dummy arguments
25: !
26:       integer :: iy , jx , nz
27:       real(4) , dimension(nz*nz,jx,iy) :: fdp
28:       real(4) , dimension(jx*nz,iy*nz) :: fsp
29:       intent (in) fdp , iy , jx , nz
30:       intent (out) fsp
31: !
32: ! Local variables
33: !
34:       integer :: i , ii , j , jj , k
35: !
36:       do j = 1 , jx*nz
37:         do i = 1 , iy*nz
38:           jj = mod(j,nz)
39:           if ( jj.eq.0 ) jj = nz
40:           ii = mod(i,nz)
41:           if ( ii.eq.0 ) ii = nz
42:           k = (jj-1)*nz + ii
43:           jj = (j+nz-1)/nz
44:           ii = (i+nz-1)/nz
45:           fsp(j,i) = fdp(k,jj,ii)
46:         end do
47:       end do
48:  
49:       end subroutine reorder
